def calcular_salario_liquido(salario_bruto, taxa_ss=0.11, taxa_irs=0.15):
    ss = salario_bruto * taxa_ss
    irs = salario_bruto * taxa_irs
    salario_liquido = salario_bruto - ss - irs
    return salario_liquido, ss, irs

# Exemplo
salario_bruto = 2000
liquido, ss, irs = calcular_salario_liquido(salario_bruto)

print(f"IRS: {irs:.2f}")

Impostos e Contribuições#

def calcular_impostos_progressivos(rendimento_anual):
    escaloes = [
        (7112, 0.145),
        (10732, 0.23),
        (20322, 0.28),
        (25075, 0.35),
        (36967, 0.37),
        (80882, 0.45),
        (float('inf'), 0.48)
    ]
    
    imposto = 0
    rendimento_restante = rendimento_anual
    
    for limite, taxa in escaloes:
        if rendimento_restante > 0:
            rendimento_tributavel = min(rendimento_restante, limite)
            imposto += rendimento_tributavel * taxa
            rendimento_restante -= rendimento_tributavel
        else:
            break
    
    return imposto

# Exemplo
rendimento_anual = 30000
imposto_devido = calcular_impostos_progressivos(rendimento_anual)

print(f"Rendimento Anual: {rendimento_anual:.2f}")
print(f"Imposto Devido: {imposto_devido:.2f}")
print(f"Taxa Efetiva de Imposto: {(imposto_devido / rendimento_anual) * 100:.2f}%")

Cálculo do IRS#

def calcular_irs(rendimento_anual, deducoes=0):
    rendimento_coletavel = rendimento_anual - deducoes
    imposto_bruto = calcular_impostos_progressivos(rendimento_coletavel)
    
    # Simulação de deduções à coleta
    deducoes_coleta = min(1000, imposto_bruto * 0.10)
    
    imposto_final = max(0, imposto_bruto - deducoes_coleta)
    
    return imposto_final

# Exemplo
rendimento_anual = 35000
deducoes = 4000

irs = calcular_irs(rendimento_anual, deducoes)

print(f"Rendimento Anual: {rendimento_anual:.2f}")
print(f"Deduções: {deducoes:.2f}")
print(f"IRS a pagar: {irs:.2f}")

def juros_simples(principal, taxa, tempo):

def juros_compostos(principal, taxa, tempo, periodos=1):
    return principal * (1 + taxa/periodos) ** (tempo * periodos)

# Exemplo
principal = 1000
taxa_anual = 0.05
anos = 5
periodos_por_ano = 12  # capitalização mensal

Valor Presente e Valor Futuro#

def valor_presente(valor_futuro, taxa, tempo):
    return valor_futuro / (1 + taxa) ** tempo

def valor_futuro(valor_presente, taxa, tempo):
    return valor_presente * (1 + taxa) ** tempo

# Exemplo
valor_futuro_desejado = 10000
taxa_anual = 0.06
anos = 10

valor_presente_necessario = valor_presente(valor_futuro_desejado, taxa_anual, anos)

print(f"Para ter {valor_futuro_desejado:.2f} daqui a {anos} anos,")

def calcular_amortizacao(principal, taxa_anual, anos):
    taxa_mensal = taxa_anual / 12
    num_pagamentos = anos * 12
    
    pagamento_mensal = (principal * taxa_mensal * (1 + taxa_mensal) ** num_pagamentos) / ((1 + taxa_mensal) ** num_pagamentos - 1)
    
    saldo = principal
    amortizacao_total = 0
    juros_total = 0
    
    print("Mês | Pagamento | Principal | Juros | Saldo")
    print("----|-----------|-----------|-------|------")
    
    for mes in range(1, num_pagamentos + 1):
        juros = saldo * taxa_mensal
        amortizacao = pagamento_mensal - juros
        saldo -= amortizacao
        
        amortizacao_total += amortizacao
        juros_total += juros
        
        print(f"{mes:4d} | {pagamento_mensal:9.2f} | {amortizacao:9.2f} | {juros:5.2f} | {saldo:6.2f}")
    
    print(f"\nTotal pago: {pagamento_mensal * num_pagamentos:.2f}")
    print(f"Total de juros: {juros_total:.2f}")

Cálculo de Prestações#

def calcular_prestacao(principal, taxa_anual, anos):
    taxa_mensal = taxa_anual / 12
    num_pagamentos = anos * 12
    
    prestacao = (principal * taxa_mensal * (1 + taxa_mensal) ** num_pagamentos) / ((1 + taxa_mensal) ** num_pagamentos - 1)
    
    return prestacao

# Exemplo

print(f"Taxa anual: {taxa_anual:.2%}")
print(f"Prazo: {anos} anos")
print(f"Prestação mensal: {prestacao:.2f}")
